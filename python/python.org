   #+BEGIN_SRC python
   # 包引入
   import fibo # 引入整个模块,不建议
   import fibo as fib # 包别名,可用于缩短包名
   from fibo import fib as fibonacci # 指定输出,建议方案
   from fibo import * # 不要这么写, 会覆盖以定义的对象
   fibo.reload() # 测试时重新加载被修改的模块

   # 以脚本方式执行模块
   # 在文件尾部添加如下代码
   if __name__ == "__main__":
       import sys
       fib(int(sys.arg[1]))
   #+END_SRC
