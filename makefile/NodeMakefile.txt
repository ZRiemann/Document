1.4 makefile variable

OBJS = main.o kbd.o command.o display.o insert.o search.o files.o utils.o
edit : $(objects)
     cc -o edit $(objects)
main.o : main.c defs.h
       cc -c main.c
...

clean 
      -rm edit $(objects)
